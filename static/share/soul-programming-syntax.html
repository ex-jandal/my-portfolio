// Varibles definitions.
var_name = 10;
var_name:: fxd = 10;
var_name:: mut u8 = uninit;

// pointers
ptr_name:: fxd *mut u8 = &var_name;

// functions or procedure here are..
// different. :)
// they are like varibles definition.
// and yes.. Soul doesn't support multi-line comments
fun_name:: proc(_ignored:: f64) -> type {
  print(_ignored);
  ret value;
}

// here is how to comment a code block.
/-fun_name:: proc(para:: u8) -> void { 
  ...
}

// struct are cool too. :)
pub Name:: struct() {
  .properity_name:: u8,
  .properity_name:: *mut str,

  pub method_name:: proc(self: mut *Self, input: *str) -> type {
    _ = input;
    ...
  }

  method_name:: proc(void) -> type {
    ...
  }
}

// Anonymise struct,
// or can used as a short-hand struct.
Name:: .{
  .properity_name: u8,
  .properity_name: *str,
}

// also anonymise struct but with no-field name,
// or just say a tuple
Name:: .{ u8, *str }

// this one is a tagged union,
// can be used as a enum or union.
pub Name:: tags() {
  .VarOne,
  .VarTwo(i32),
  .VarTree .{ .x:: f32, .y:: f32},
  ...
}

// Optional values are just tags in Soul programming,
// also Result.
var_name:: mut Opt(u8) = .null;

// Soul Support error union or tags.
fut_err:: error {
  .Stopped,
  // quoted-identifiers to use a keyword or wired text as a varible or function name
  .#"can't get the value",
  .SomthingElseGoesWrong,
}

// tags with generics and method support like Zig and Rust (Maybe).
pub Fut:: tags(T:: u8 | f32 | *mut str) {
  rsv(T),
  rjc(fut_err),

  pub await:: proc(self:: *mut Self) -> Result(T) {
    // .. Wating logic
    if true {
      ret .ok(T);
    } else {
      self.* = .rjc(fut_err.#"can't get the value");
      ret self.*;
    }
  }
}

async fun_name:: proc(input:: f64) -> Fut<type> {
  if input >= 0 {
    ret .rjc(.SomthingElseGoesWrong);
  }
  ret .rsv(input + 42.0);
}

main:: proc(init: std.Init) -> Result<u8> {
  fut_handler:: mut Fut = fun_name(42.0);
  fut_handler.await();
  ret 0;
}
